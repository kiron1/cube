82 static GLuint make_shader(GLenum type, std::string const& source)
83 {
84  const GLchar* src[] = { source.c_str() };
85  const GLint srclen[] = { static_cast<GLint>(source.size()) };
86  GLuint shader;
87  GLint shader_ok;
88 
89  shader = glCreateShader(type);
90  glShaderSource(shader, 1, src, srclen);
91  glCompileShader(shader);
92 
93  glGetShaderiv(shader, GL_COMPILE_STATUS, &shader_ok);
94  if (!shader_ok) {
95  std::cerr
96  << "Failed to compile shader:\n"
97  << info_log(shader, glGetShaderiv, glGetShaderInfoLog)
98  << std::endl;
99  glDeleteShader(shader);
100  return 0;
101  }
102  return shader;
103 }

107 {

124  }

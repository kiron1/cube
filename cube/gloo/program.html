Class program

cube::gloo::program — OpenGL shader program, linking the shaders and binding the variables.

Synopsis

// In header: </home/ion/src/cube/include/cube/gloo/program.hpp>


class program : public cube::gloo::globject< program, with_handle > {
public:
  // types
  typedef unspecified attribute_table;
  typedef unspecified uniform_table;  

  enum draw_mode { points = = GL_POINTS, lines = = GL_LINES, 
                   line_strip = = GL_LINE_STRIP, line_loop = = GL_LINE_LOOP, 
                   triangles = = GL_TRIANGLES, 
                   triangle_strip = = GL_TRIANGLE_STRIP, 
                   triangle_fan = = GL_TRIANGLE_FAN };

  // construct/copy/destruct
  program();
  program(vertex_shader, fragment_shader);
  program(std::string, std::string);

  // public member functions
  void attach(vertex_shader const &);
  void attach(fragment_shader const &);
  void draw(draw_mode);
  void draw(draw_mode, element_index_base &);

  // private member functions
  void attach(attribute_base *);
  void attach(uniform_base *);
  unspecified variable(attribute_base *);
  unspecified variable(uniform_base *);
  void detach(attribute_base *);
  void detach(uniform_base *);
  attribute_table active_attributes();
  uniform_table active_uniforms();
  void create();
  void enable();
  void disable();
  void destroy();
};
